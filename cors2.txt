const AirscanProcessCard = ({
  process,
  deviceCount,
  onClick,
}: {
  process: ProcessList;
  deviceCount: number;
  onClick: ({ id }: { id: number }) => void;
}) => {
  // Determine status and color
  const isCompleted = process.timerWindow === "00:00:00";
  const isInProgress = !isCompleted;
  const bottomBarClass = isCompleted ? "process-card-bottom-green" : "process-card-bottom-orange";

  return (
    <div
      className="process-card-view cursor-pointer"
      onClick={() => onClick({ id: process.id })}
    >
      <div className="process-card-content">
        <div className="process-card-title">
          <strong>Process: {process.id}</strong>
        </div>
        <div className="process-card-detail">
          <strong>Device Count:</strong> {deviceCount}
        </div>
        <div className="process-card-detail">
          <strong>Time remaining :</strong>{" "}
          <span className={isCompleted ? "process-time-green" : "process-time-orange"}>{process.timerWindow}</span>
        </div>
        <div className="process-card-link">
          <a
            href="#"
            className="process-card-hyperlink"
            onClick={e => { e.stopPropagation(); /* handle link click if needed */ }}
          >
            Show calibrated Value
          </a>
        </div>
      </div>
      <div className={`process-card-bottom-bar ${bottomBarClass}`} />
    </div>
  );
};
